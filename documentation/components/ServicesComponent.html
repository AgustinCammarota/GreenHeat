import { ChangeDetectionStrategy, Component, signal, WritableSignal } from '@angular/core';

@Component({
  selector: 'app-services',
  standalone: true,
  imports: [
    CardComponent,
    TitleComponent
  ],
  templateUrl: './services.component.html',

  ]);
