import { Brand, Presentation } from '@home/interfaces';

@Component({
  selector: 'app-clients',
  standalone: true,
  imports: [
    TitleComponent,
    BrandComponent
  ],
  templateUrl: './clients.component.html',
  styleUrl: './clients.component.scss',
  changeDetection: ChangeDetectionStrategy.OnPush
})

 * Client Section

  ]);
}
